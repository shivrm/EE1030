    \begin{align*}
    \psi_1 = \varphi_0 - 2\varphi_1 + 3\varphi_2 \\
    \psi_2 = \varphi_0 - \varphi_1 + \alpha\varphi_2
    \end{align*}

    where $\alpha$ is a constant.

            \item $\frac{3\hbar\omega}{2}$
            \item $3\hbar\omega$
            \item $\frac{9\hbar\omega}{2}$
        \end{enumerate}
    \end{multicols}
